% = é o modulo da divisão.
++ = acrescenta 1 a cada execução
-- = diminui 1 a cada execução

let raio = 10
console.log(++raio)
11
console.log(--raio)
10

e(&&) = todas tem que ser verdadeira para dar true.

raio > 2
verdadeiro
raio > 2 && raio > 11
falso
raio > 2 && raio < 11
verdadeiro

ou(||) = uma tem que ser verdadeira para dar true.

raio < 2
falso
raio > 2 || raio > 11
verdadeiro
raio > 2 || raio < 11
verdadeiro

não(!) = transforma oq for verdadeiro em falso e vice versa.

!raio < 2
verdadeiro
!raio > 2 || raio > 11
falso
!raio > 2 || raio < 11
falso
